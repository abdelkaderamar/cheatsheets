\cppitem{Read a json file and convert a list of json objects to list of objects}
\begin{minted}
  [frame=single, rulecolor=blue, framesep=1mm, baselinestretch=1, fontsize=\footnotesize]{cpp}
with open(filename, 'r') as f:
        data = json.load(f)
        if 'result_data' in data:
            stocks=[Stock(**d) for d in data['result_data']]

\end{minted}

\cppitem{Parse a string as a json object}
\begin{minted}
  [frame=single, rulecolor=blue, framesep=1mm, baselinestretch=1, fontsize=\footnotesize]{cpp}
data = json.loads(script.text)
instrument = data['custom']['instrument']

\end{minted}

\cppitem{Print a json object}
\begin{minted}
  [frame=single, rulecolor=blue, framesep=1mm, baselinestretch=1, fontsize=\footnotesize]{cpp}
o =   { "name": "AZZ Incorporated", "ticker": "AZZ", "capitalisation": "1037603000" }
print(json.dumps(o, indent=2))

\end{minted}
