{\centering\section{Strings}}

\cppitem{Remove newlines from string}
\begin{minted}
  [frame=single, rulecolor=blue, framesep=1mm, baselinestretch=1, fontsize=\footnotesize]{cpp}
${str//[$'\t\r\n ']} 

\end{minted}

\cppitem{Replace some characters by another}
\begin{minted}
  [frame=single, rulecolor=blue, framesep=1mm, baselinestretch=1, fontsize=\footnotesize]{cpp}
mod=${orig//[xyz]/_}

\end{minted}

\cppitem{Delete line containing a pattern}
\begin{minted}
  [frame=single, rulecolor=blue, framesep=1mm, baselinestretch=1, fontsize=\footnotesize]{cpp}
sed 's/pattern/d' file.txt

\end{minted}

\cppitem{Check if a string starts with a pattern}
\begin{minted}
  [frame=single, rulecolor=blue, framesep=1mm, baselinestretch=1, fontsize=\footnotesize]{cpp}
if [[ "$var" == ${pattern}* ]]
# or
if [[ "$var" =~ ^${pattern}.* ]]

\end{minted}

{\centering\section{Arrays}}

\cppitem{Array split}
\begin{minted}
  [frame=single, rulecolor=blue, framesep=1mm, baselinestretch=1, fontsize=\footnotesize]{cpp}
arr=("a" "b" "c" "d")
team_one=("${arr[@]:0:2}")
echo "${arr[@]}"
# a b c d
echo "${team_one[@]}"
# a b

\end{minted}

\cppitem{Array of string from multi-line string}
\begin{minted}
  [frame=single, rulecolor=blue, framesep=1mm, baselinestretch=1, fontsize=\footnotesize]{cpp}
mapfile -t arr <<< "$str"

\end{minted}
