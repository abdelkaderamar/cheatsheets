\cppitem{Interleave lines of two text files}
\begin{minted}
  [frame=single, rulecolor=blue, framesep=1mm, baselinestretch=1, fontsize=\footnotesize]{cpp}
paste -d '\n' file1 file2

\end{minted}

\cppitem{exiftool - Renaming images}
\begin{minted}
  [frame=single, rulecolor=blue, framesep=1mm, baselinestretch=1, fontsize=\footnotesize]{cpp}
# Renames all images in directory "DIR" according to the individual file's creation date in the form "YYYYmmdd\_HHMMSS.ext".
exiftool "-FileName<CreateDate"
     -d "%Y%m%d_%H%M%S.%%e" DIR

\end{minted}

\cppitem{List package content (before installing it)}
\begin{minted}
  [frame=single, rulecolor=blue, framesep=1mm, baselinestretch=1, fontsize=\footnotesize]{cpp}
apt-file update
apt list package_name

\end{minted}

\cppitem{List (installed) package content}
\begin{minted}
  [frame=single, rulecolor=blue, framesep=1mm, baselinestretch=1, fontsize=\footnotesize]{cpp}
dpkg -L package_name

\end{minted}

\cppitem{Which package provides a file}
\begin{minted}
  [frame=single, rulecolor=blue, framesep=1mm, baselinestretch=1, fontsize=\footnotesize]{cpp}
dpkg -S file_absolute_path

\end{minted}

\cppitem{Find which package provides a file (not installed)}
\begin{minted}
  [frame=single, rulecolor=blue, framesep=1mm, baselinestretch=1, fontsize=\footnotesize]{cpp}
apt-file search file

\end{minted}

\cppitem{exiftool - Moving images}
\begin{minted}
  [frame=single, rulecolor=blue, framesep=1mm, baselinestretch=1, fontsize=\footnotesize]{cpp}
Moves all images originally in directory "DIR" into a directory hierarchy organized by year/month/day
exiftool "-Directory<DateTimeOriginal" -d "%Y/%m/%d" DIR

\end{minted}

\cppitem{Mount a disk}
\begin{minted}
  [frame=single, rulecolor=blue, framesep=1mm, baselinestretch=1, fontsize=\footnotesize]{cpp}
udisksctl mount -b /dev/sdg1

\end{minted}

\cppitem{Rename images in directory ``DIR'' into a directory hierarchy organized by ``Model''/year-month-day/date\_time.extension and preserve file date}
\begin{minted}
  [frame=single, rulecolor=blue, framesep=1mm, baselinestretch=1, fontsize=\footnotesize]{cpp}
exiftool -o . '-FileName<${model}/${CreateDate}' -d '%Y-%m-%d/%Y%m%d_%H%M%S.%%e' . -P

\end{minted}

\cppitem{Change or remove Camera Model Name tag from an image}
\begin{minted}
  [frame=single, rulecolor=blue, framesep=1mm, baselinestretch=1, fontsize=\footnotesize]{cpp}
exiftool -'Model'= 20190607_213819.jpg

\end{minted}
