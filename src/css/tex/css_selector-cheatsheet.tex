\cppitem{Get element with a class}
\begin{minted}
  [frame=single, rulecolor=blue, framesep=1mm, baselinestretch=1, fontsize=\footnotesize]{cpp}
element.class
Example: div.product to get all divs of class product

\end{minted}

\cppitem{Get the text of an element}
\begin{minted}
  [frame=single, rulecolor=blue, framesep=1mm, baselinestretch=1, fontsize=\footnotesize]{cpp}
element::text
Example: div::text

\end{minted}

\cppitem{Get the text of an element with a class}
\begin{minted}
  [frame=single, rulecolor=blue, framesep=1mm, baselinestretch=1, fontsize=\footnotesize]{cpp}
element.class::text
Example: div.product-subtitle::text

\end{minted}

\cppitem{Get the elements depending on the parent}
\begin{minted}
  [frame=single, rulecolor=blue, framesep=1mm, baselinestretch=1, fontsize=\footnotesize]{cpp}
element>element
Example: div>img select img elements where the parent is a div

\end{minted}

\cppitem{Get the attribute depending on the parent class}
\begin{minted}
  [frame=single, rulecolor=blue, framesep=1mm, baselinestretch=1, fontsize=\footnotesize]{cpp}
element.class>element::attr(attr_name)
Example: div.thumbnail>img::attr(data-src) select data-src attribute of img 
elements where the parent is a div with the class thumbnail

\end{minted}
