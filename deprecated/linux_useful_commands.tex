\linuxitem{exiftool - Renaming images}
Renames all images in directory "DIR" according to the individual file's creation date in the form "YYYYmmdd\_HHMMSS.ext".
\begin{minted}
  [frame=single, rulecolor=blue, framesep=1mm, baselinestretch=1, fontsize=\footnotesize]{bash}
  exiftool "-FileName<CreateDate" \
     -d "%Y%m%d_%H%M%S.%%e" DIR
\end{minted}

\begin{center}{\section{Packages}}\end{center}

\linuxitem{List package content (before installing it)}
\begin{minted}
  [frame=single, rulecolor=blue, framesep=1mm, baselinestretch=1, fontsize=\footnotesize]{bash}
apt-file update
apt list <package name>
\end{minted}

\linuxitem{List (installed) package content}
\begin{minted}
  [frame=single, rulecolor=blue, framesep=1mm, baselinestretch=1, fontsize=\footnotesize]{bash}
dpkg -L <package name>
\end{minted}

\linuxitem{Which package provides a file}
\begin{minted}
  [frame=single, rulecolor=blue, framesep=1mm, baselinestretch=1, fontsize=\footnotesize]{bash}
dpkg -S <file absolute path>
\end{minted}

\linuxitem{Find which package provides a file (not installed)}
\begin{minted}
  [frame=single, rulecolor=blue, framesep=1mm, baselinestretch=1, fontsize=\footnotesize]{bash}
apt-file search <file>
\end{minted}

\linuxitem{exiftool - Moving images}
Moves all images originally in directory "DIR" into a directory hierarchy organized by year/month/day
\begin{minted}
  [frame=single, rulecolor=blue, framesep=1mm, baselinestretch=1, fontsize=\footnotesize]{bash}
exiftool "-Directory<DateTimeOriginal" -d "%Y/%m/%d" DIR
\end{minted}

udisksctl mount -b /dev/sdg1

\linuxitem{Feature}
Rename images in directory ``DIR'' into a directory hierarchy organized by ``Model''/year-month-day/date\_time.extension
\begin{minted}
  [frame=single, rulecolor=blue, framesep=1mm, baselinestretch=1, fontsize=\footnotesize]{tex}
  exiftool -o . '-FileName<${model}/${CreateDate}' -d '%Y-%m-%d/%Y%m%d_%H%M%S.%%e' .
\end{minted}

\linuxitem{Feature}
Rename images in directory ``DIR'' into a directory hierarchy organized by ``Model''/year-month-day/date\_time.extension and preserve file date
\begin{minted}
  [frame=single, rulecolor=blue, framesep=1mm, baselinestretch=1, fontsize=\footnotesize]{tex}
  exiftool -o . '-FileName<${model}/${CreateDate}' -d '%Y-%m-%d/%Y%m%d_%H%M%S.%%e' . -P
\end{minted}

\linuxitem{Feature}
Change or remove Camera Model Name tag from an image
\begin{minted}
  [frame=single, rulecolor=blue, framesep=1mm, baselinestretch=1, fontsize=\footnotesize]{tex}
exiftool -'Model'= 20190607_213819.jpg
\end{minted}
