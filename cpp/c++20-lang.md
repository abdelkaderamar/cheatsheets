title: using enum

  NONE,
  DAILY,
  WEEKLY,
  MONTHLY
};
switch (frequency) {
  using enum frequency_type;
  {
    case NONE: break;
    case DAILY: break;
    case WEEKLY: break;
    case MONTHLY: break;
  }
}
